@echo off
echo Starting GuardLink Backend and Frontend...

REM Optional first argument can be a backend base URL. Example:
REM run_both.bat https://my-live-backend.example.com
set "BACKEND_URL=http://127.0.0.1:5000"
if not "%~1"=="" (
    set "BACKEND_URL=%~1"
)
if not "%BACKEND_URL:~-4%"=="/api" (
    set "BACKEND_URL=%BACKEND_URL%/api"
)

REM Start Backend in background
start cmd /c "cd /d d:\accident\backend && .\venv\Scripts\python.exe run_server.py"

REM Wait a moment for backend to initialize
timeout /t 5 /nobreak > nul

REM Start Frontend with live backend API URL
cd /d d:\accident\frontend
set "VITE_API_URL=%BACKEND_URL%"
npm run dev -- --host 0.0.0.0 --port 5173

echo Both services started. Backend: %BACKEND_URL% Frontend: http://127.0.0.1:5173
